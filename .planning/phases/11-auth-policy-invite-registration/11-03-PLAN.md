    # --- Invite Creation Tests ---

    def test_create_invite_as_admin_succeeds(client, admin_cookie):
        """Admin can create an invite link."""
        response = client.post("/api/auth/invite", cookies=admin_cookie)
        assert response.status_code == 201
        data = response.json()
        assert "token" in data
        assert "invite_url" in data
        assert "invite_url" in data and "invite=" in data["invite_url"]
        assert data["is_used"] is False


    def test_create_invite_as_non_admin_fails(client, session):
        """Non-admin cannot create invite links."""
        # Create a regular user
        user = User(
            username="regularuser",
            email="user@test.com",
            password_hash=hash_password("UserPass1"),
            role="user"
        )
        session.add(user)
        session.commit()

        login_resp = client.post("/api/auth/login", json={"email": "user@test.com", "password": "UserPass1"})
        assert login_resp.status_code == 200

        response = client.post("/api/auth/invite", cookies=login_resp.cookies)
        assert response.status_code == 403


    # --- Invite Registration Gate Tests ---

    def test_register_without_invite_rejected(client):
        """Registration without invite token is rejected with HTTP 400."""
        response = client.post("/api/auth/register", json={
            "username": "newuser",
            "email": "newuser@test.com",
            "password": "SecurePass1"
        })
        assert response.status_code == 400
        assert "undangan" in response.json()["detail"].lower()


    def test_register_with_invalid_invite_rejected(client):
        """Registration with a non-existent invite token is rejected."""
        response = client.post("/api/auth/register", json={
            "username": "newuser",
            "email": "newuser@test.com",
            "password": "SecurePass1",
            "invite_token": "00000000-0000-0000-0000-000000000000"
        })
        assert response.status_code == 400


    def test_register_with_valid_invite_succeeds(client, valid_invite_token):
        """Registration with a valid invite token succeeds."""
        response = client.post("/api/auth/register", json={
            "username": "newuser",
            "email": "newuser@test.com",
            "password": "SecurePass1",
            "invite_token": valid_invite_token
        })
        assert response.status_code == 201
        data = response.json()
        assert data["email"] == "newuser@test.com"


    def test_invite_is_single_use(client, valid_invite_token):
        """Invite token cannot be reused after first registration."""
        # First registration succeeds
        client.post("/api/auth/register", json={
            "username": "user1",
            "email": "user1@test.com",
            "password": "SecurePass1",
            "invite_token": valid_invite_token
        })
        # Second registration with same token fails
        response = client.post("/api/auth/register", json={
            "username": "user2",
            "email": "user2@test.com",
            "password": "SecurePass1",
            "invite_token": valid_invite_token
        })
        assert response.status_code == 400
        assert "sudah digunakan" in response.json()["detail"]


    # --- Password Policy Tests ---

    def test_register_with_no_uppercase_rejected(client, valid_invite_token):
        """Password without uppercase letter is rejected with HTTP 422."""
        response = client.post("/api/auth/register", json={
            "username": "newuser",
            "email": "newuser@test.com",
            "password": "securepass1",  # no uppercase
            "invite_token": valid_invite_token
        })
        assert response.status_code == 422


    def test_register_with_no_digit_rejected(client, admin_cookie, session):
        """Password without a digit is rejected with HTTP 422."""
        invite_resp = client.post("/api/auth/invite", cookies=admin_cookie)
        token = invite_resp.json()["token"]
        response = client.post("/api/auth/register", json={
            "username": "newuser",
            "email": "newuser@test.com",
            "password": "SecurePassword",  # no digit
            "invite_token": token
        })
        assert response.status_code == 422


    def test_register_with_short_password_rejected(client, admin_cookie):
        """Password shorter than 8 characters is rejected."""
        invite_resp = client.post("/api/auth/invite", cookies=admin_cookie)
        token = invite_resp.json()["token"]
        response = client.post("/api/auth/register", json={
            "username": "newuser",
            "email": "newuser@test.com",
            "password": "Ab1",  # too short
            "invite_token": token
        })
        assert response.status_code == 422


    # --- Account Lockout Tests ---

    def test_account_lockout_after_5_failed_attempts(client, admin_user):
        """Account is locked after 5 failed login attempts from the same IP."""
        # Clear any existing failed attempts
        _failed_attempts.clear()

        # 5 failed attempts (should NOT be locked yet)
        for i in range(5):
            resp = client.post("/api/auth/login", json={
                "email": "admin@test.com",
                "password": "WrongPassword"
            })
            assert resp.status_code == 400, f"Attempt {i+1} should fail with 400, got {resp.status_code}"

        # 6th attempt should be locked
        resp = client.post("/api/auth/login", json={
            "email": "admin@test.com",
            "password": "WrongPassword"
        })
        assert resp.status_code == 423, f"6th attempt should be locked (423), got {resp.status_code}"
        assert "dikunci" in resp.json()["detail"].lower()


    def test_successful_login_clears_failed_attempts(client, admin_user):
        """Successful login clears the failed attempt counter."""
        _failed_attempts.clear()

        # 3 failed attempts
        for _ in range(3):
            client.post("/api/auth/login", json={"email": "admin@test.com", "password": "Wrong"})

        # Successful login clears counter
        resp = client.post("/api/auth/login", json={"email": "admin@test.com", "password": "AdminPass1"})
        assert resp.status_code == 200

        # Now 3 more failures should NOT trigger lockout (counter was reset)
        for _ in range(3):
            client.post("/api/auth/login", json={"email": "admin@test.com", "password": "Wrong"})

        resp = client.post("/api/auth/login", json={"email": "admin@test.com", "password": "Wrong"})
        assert resp.status_code == 400  # Not 423, counter was reset
