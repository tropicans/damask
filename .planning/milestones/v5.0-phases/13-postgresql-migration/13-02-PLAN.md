### Task 1 — Create `docker-compose.prod.yml`

**File:** `docker-compose.prod.yml` (project root)
**Action:** CREATE NEW

```yaml
version: '3.8'

services:
  db:
    image: postgres:16-alpine
    restart: unless-stopped
    environment:
      POSTGRES_DB: securedata
      POSTGRES_USER: securedata_user
      POSTGRES_PASSWORD: ${POSTGRES_PASSWORD}
    volumes:
      - postgres_data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U securedata_user -d securedata"]
      interval: 10s
      timeout: 5s
      retries: 5
      start_period: 30s

  backend:
    build: ./backend
    restart: unless-stopped
    ports:
      - "8000:8000"
    environment:
      - DATABASE_URL=postgresql://securedata_user:${POSTGRES_PASSWORD}@db:5432/securedata
      - JWT_SECRET_KEY=${JWT_SECRET_KEY}
      - CORS_ALLOWED_ORIGINS=${CORS_ALLOWED_ORIGINS}
      - COOKIE_SECURE=true
      - FRONTEND_URL=${FRONTEND_URL}
    depends_on:
      db:
        condition: service_healthy

  frontend:
    build: ./frontend
    restart: unless-stopped
    ports:
      - "5173:5173"
    environment:
      - VITE_API_URL=${VITE_API_URL}
    depends_on:
      - backend

volumes:
  postgres_data:
```

**Notes:**
- Sensitive values (`POSTGRES_PASSWORD`, `JWT_SECRET_KEY`) are referenced via `${VAR}` from a `.env` file at project root (not committed to git).
- `db` service uses `postgres:16-alpine` for a small, production-ready image.
- `backend` has `depends_on: db: condition: service_healthy`, so it only starts after `pg_isready` passes. This works in conjunction with the retry loop added in Plan 13-01.
- `postgres_data` named volume ensures data persists across container restarts.

### Task 2 — Create `backend/.env.example`

**File:** `backend/.env.example`
**Action:** CREATE NEW

```dotenv
# =============================================================================
# SecureData Web — Backend Environment Variables
# =============================================================================
# Copy this file to backend/.env and fill in your values.
# NEVER commit backend/.env to version control.

# ---------------------------------------------------------------------------
# Database
# ---------------------------------------------------------------------------

# Development (SQLite — zero-config, default):
DATABASE_URL=sqlite:///datamask.db

# Production (PostgreSQL):
# DATABASE_URL=postgresql://securedata_user:your_password@localhost:5432/securedata

# CI / Integration tests against a real PostgreSQL instance:
# TEST_DATABASE_URL=postgresql://securedata_user:your_password@localhost:5432/testdb

# ---------------------------------------------------------------------------
# Security
# ---------------------------------------------------------------------------

# Generate with: python -c "import secrets; print(secrets.token_hex(32))"
JWT_SECRET_KEY=change-me-in-production

JWT_ALGORITHM=HS256

# Access token lifetime in minutes (default 1440 = 24 hours)
ACCESS_TOKEN_EXPIRE_MINUTES=1440

# Set to true in production (requires HTTPS)
COOKIE_SECURE=false

# ---------------------------------------------------------------------------
# CORS
# ---------------------------------------------------------------------------

# JSON array of allowed frontend origins
CORS_ALLOWED_ORIGINS=["http://localhost:5173"]

# ---------------------------------------------------------------------------
# Application
# ---------------------------------------------------------------------------

PORT=8000

# Base URL used for constructing invite registration links
FRONTEND_URL=http://localhost:5173

# Invite link expiry in hours (default 48)
INVITE_EXPIRE_HOURS=48
```
